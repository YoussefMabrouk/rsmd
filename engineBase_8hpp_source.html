75//
76// execute the command (+ cmdline options) given by args with piped input (parent->child)
77//
78template<typename... Args>
79void EngineBase::execute( std::string& pipeIn, const char* cmd, Args&& ... args )
80{
81 int status;
82 int childIn[2], childOut[2]; // for piping input/output of the child process
83 pid_t child_pid;
84 char buffer[101] = {0};
85 ssize_t readResult {0};
86 std::string pipeOut {};
87
88 // about the pipes:
89 // the first integer in the respective fd (file descriptor) array (element 0) is set up and opened for reading,
90 // while the second integer (element 1) is set up and opened for writing.
91 // visually speaking, the output of fd[1] becomes the input for fd[0].
92 // once again, all data traveling through the pipe moves through the kernel.
93
94 // some verbosity:
95 std::stringstream stream {};
96 stream << "[EngineBase::execute()] running:";
97 if( ! pipeIn.empty() )
98 {
99 std::string tmp {pipeIn};
100 tmp.erase(std::remove(tmp.begin(), tmp.end(), '\n'), tmp.end());
101 stream << " " << tmp << " |";
102 }
103 using expander = int[];
104 (void) expander {0, (void(stream << ' ' << std::forward<const char*>(args)),0)...};
105 rsmdDEBUG( stream.str() );
106
107 // creating the pipes
108 if( pipe(childIn) < 0) // failure in creating a pipe
109 rsmdCRITICAL( "failure in creating a pipe")
110 if( pipe(childOut) < 0) // failure in creating a pipe
111 rsmdCRITICAL( "failure in creating a pipe" );
112
113 // try forking a new process
114 child_pid = fork();
115 // about fork():
116 // returns -1 in case of failure,
117 // 0 to the newly created child process and
118 // child_pid (positive value) to the parent
119 switch( child_pid )
120 {
121 case -1: // fork failed
122 rsmdCRITICAL( "fork failed" );
123 break;
124
125 case 0: // fork successful, this section is only entered from within child
126 // replace standard input with input part of pipe and close unused other half
127 dup2(childIn[READ_FD], READ_FD); // dup2() includes closing of this file descriptor!
128 close(childIn[WRITE_FD]);
129
130 // replace standard output with output part of pipe and close unused other half
131 dup2(childOut[WRITE_FD], STDOUT_FILENO);
132 dup2(childOut[WRITE_FD], STDERR_FILENO);
133 close(childOut[READ_FD]);
134
135 // execute program:
136 execlp( cmd, std::forward<const char*>(args)..., NULL ); // the program should receive its own command as argv[0]
137
138 // should't return, so exit here
139 std::exit(EXIT_FAILURE);
140
141 break;
142
143 default: // fork successful, this section is only entered from within parent
144 // write to childIn[1] (which will become stdin for the child) and close both sides of the pipe afterwards
145 close( childIn[READ_FD] );
146 write( childIn[WRITE_FD], pipeIn.c_str(), (strlen( pipeIn.c_str() )+1) );
147 close( childIn[WRITE_FD]) ;
148
149 // wait for state changes in the child (aka for it to return)
150 // and handle exit status or any signals correctly
151 do
152 {
153 wait( &status );
154 if( WIFEXITED(status) )
155 {
156 rsmdDEBUG( "[EngineBase::execute()] " << "exited: status = " << WEXITSTATUS(status) );
157 }
158 else if( WIFSIGNALED(status) )
159 {
160 rsmdWARNING( "[EngineBase::execute()] " << "killed by signal " << WTERMSIG(status) );
161 std::raise( WTERMSIG(status) );
162 }
163 else if( WIFSTOPPED(status) )
164 {
165 rsmdWARNING( "[EngineBase::execute()] " << "stopped by signal " << WSTOPSIG(status) );
166 std::raise( WSTOPSIG(status) );
167 }
168 else if( WIFCONTINUED(status) )
169 {
170 rsmdWARNING( "[EngineBase::execute()] " << "continued" );
171 }
172 } while( !WIFEXITED(status) && !WIFSIGNALED(status) );
173
174 // close writing part of file descriptor
175 close( childOut[WRITE_FD] );
176
177 // get output from reading part of file descriptor
178 while( (readResult = read(childOut[READ_FD], buffer, sizeof(buffer)-1)) > 0 )
179 {
180 buffer[readResult] = 0;
181 pipeOut.append( buffer );
182 }
183
184 // finally also close reading part
185 close(childOut[READ_FD]);
186
187 // spill process output in case anything went wrong
188 if( WEXITSTATUS(status) != 0 || WIFSIGNALED(status) || WIFSTOPPED(status) || WIFCONTINUED(status) )
189 {
190 rsmdWARNING( "process output was: \n" << pipeOut );
191 }
192
193 // throw exception in case exited with status != 0
194 if( WEXITSTATUS(status) != 0 )
195 {
196 throw std::runtime_error("something went wrong in child process execution");
197 }
198
199 break;
200 } // end of switch()
201}
